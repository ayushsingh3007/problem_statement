You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. Input:abcadeecfb Output:abcdef"

ans- const inputString = 'abcadeecfb';
const outputString = removeDuplicates(inputString);
 function removeDuplicates(str) {
  let result = '';
  let charcter = {};

  for (let i = 0; i < str.length; i++) {
    const ch = str[i];
    if (!charcter[ch]) {
      result += ch;
      charcter[ch] = true;
    }
  }

  return result;
}

ans-const string = "abcadeecfb";

  function countAlphabets(string) {
  const alphabetCount = new Map();

  for (let i = 0; i < string.length; i++) {
    const char = string[i];
    if (/[a-zA-Z]/.test(char)) {
      if (alphabetCount.has(char)) {
        alphabetCount.set(char, alphabetCount.get(char) + 1);
      } else {
        alphabetCount.set(char, 1);
      }
    }
  }

  for (const [char, count] of alphabetCount) {
    console.log(`${char}=${count}`);
  }
}
